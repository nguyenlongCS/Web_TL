db.getCollectionNames().forEach(c => {
  print(`\n[${c}]`);
  db.getCollection(c).aggregate([
    { $project: { f: { $objectToArray: "$$ROOT" } } },
    { $unwind: "$f" },
    { $group: { _id: "$f.k", t: { $addToSet: { $type: "$f.v" } } } },
    { $sort: { _id: 1 } }
  ]).forEach(f => print(`- ${f._id}: ${f.t.join(', ')}`));
});
